# Problem: Remove Duplicates from Sorted Array - https://leetcode.com/problems/remove-duplicates-from-sorted-array/

/**
 * @param {number[]} nums
 * @return {number}
 */
var removeDuplicates = function (nums) {
    if (nums.length === 0) return 0;

    let uniqueIndex = 0; // Points to the last unique element

    for (let i = 1; i < nums.length; i++) {
        if (nums[i] !== nums[uniqueIndex]) {
            uniqueIndex++; // Move to the next position for a unique value
            nums[uniqueIndex] = nums[i]; // Update the position with the new unique value
        }
    }

    // Fill the remaining elements with '_'
    for (let i = uniqueIndex + 1; i < nums.length; i++) {
        nums[i] = '_';
    }

    return uniqueIndex + 1; // Number of unique elements
};
